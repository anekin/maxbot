## 五、BS点计算Python代码

```python
import pandas as pd
import numpy as np

def calculate_bs_score(df):
    """
    计算BS点评分
    df包含: close, volume, high, low
    """
    score = 0
    
    # 1. MA均线 (20%)
    df['MA5'] = df['close'].rolling(5).mean()
    df['MA20'] = df['close'].rolling(20).mean()
    if df['MA5'].iloc[-1] > df['MA20'].iloc[-1]:
        ma_score = 20
    else:
        ma_score = -20
    
    # 2. MACD (20%)
    exp1 = df['close'].ewm(span=12).mean()
    exp2 = df['close'].ewm(span=26).mean()
    df['DIF'] = exp1 - exp2
    df['DEA'] = df['DIF'].ewm(span=9).mean()
    if df['DIF'].iloc[-1] > df['DEA'].iloc[-1]:
        macd_score = 20
    else:
        macd_score = -20
    
    # 3. RSI (15%)
    delta = df['close'].diff()
    gain = delta.where(delta > 0, 0).rolling(14).mean()
    loss = (-delta.where(delta < 0, 0)).rolling(14).mean()
    rs = gain / loss
    rsi = 100 - (100 / (1 + rs))
    if rsi.iloc[-1] < 30:
        rsi_score = 15
    elif rsi.iloc[-1] > 70:
        rsi_score = -15
    else:
        rsi_score = 0
    
    # 4. 布林带 (15%)
    df['MA20'] = df['close'].rolling(20).mean()
    df['std'] = df['close'].rolling(20).std()
    df['upper'] = df['MA20'] + 2 * df['std']
    df['lower'] = df['MA20'] - 2 * df['std']
    
    close = df['close'].iloc[-1]
    lower = df['lower'].iloc[-1]
    upper = df['upper'].iloc[-1]
    
    if close < lower * 1.02:  # 接近或跌破下轨
        bb_score = 15
    elif close > upper * 0.98:  # 接近或突破上轨
        bb_score = -15
    else:
        bb_score = 0
    
    # 5. 成交量 (15%)
    vol_avg = df['volume'].rolling(20).mean().iloc[-1]
    vol_current = df['volume'].iloc[-1]
    if vol_current > vol_avg * 1.2:
        vol_score = 15
    else:
        vol_score = -5
    
    # 综合评分
    total = (ma_score * 0.2 + macd_score * 0.2 + 
             rsi_score * 0.15 + bb_score * 0.15 + vol_score * 0.15)
    
    # BS点判断
    if total >= 40:
        signal = 'B'  # 买入
    elif total <= -40:
        signal = 'S'  # 卖出
    else:
        signal = 'H'  # 持有
    
    return {
        'signal': signal,
        'score': total,
        'ma': ma_score,
        'macd': macd_score,
        'rsi': rsi_score,
        'bollinger': bb_score,
        'volume': vol_score
    }

# 使用示例
```
